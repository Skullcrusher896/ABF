
aba_htm_chyhyryn_order = {
    icon = GFX_aba_order_chyhyrn
    trigger = {
		has_government_attribute = enables_aba_htm_chyhyryn_order
    }
    cost = 50
    cost_type = adm_power
    color = { 9 98 30 }
    per_province_effect = {
        add_base_tax = 1
        if = {
            limit = {
                owner = {
                    has_country_flag = aba_htm_regiment_district_buff_flag
                }
            }
            add_province_modifier = {
                name = aba_htm_efficient_colonels
                duration = -1
            }
        }
    }
    per_province_abandon_effect = {
       add_base_tax = -1
        if = {
            limit = {
                owner = {
                    NOT = {
                        has_country_flag = aba_htm_regiment_district_buff_flag
                    }
                }
                has_province_modifier = aba_htm_efficient_colonels
            }
           remove_province_modifier = aba_htm_efficient_colonels
        }
    }
    modifier = {
        local_governing_cost = -0.10
        local_tax_modifier = 0.10
        min_local_autonomy = 5
    }
    ai_priority = {
        factor = 1
    }
    localization = aba_htm_regimental_district
}


latin_sepulchre = {
    icon = GFX_holy_order_sepulchre
    trigger = {
		OR = {
			AND = {
				tag = LAT
				religion = catholic
			}
            has_government_attribute = enables_latin_sepulchre_holy_order
		}
    }
    cost = 50
    cost_type = adm_power
    color = { 111 111 111 }
    per_province_effect = {
        add_base_tax = 1
        if = {
            limit = {
                owner = {
                    has_country_flag = lat_upgrade_holy_orders
                }
            }
            random_list = {
                50 = {
                    ab_add_or_upgrade_road_building = yes
                }
                50 = {
                }
            }
        }
    }
    per_province_abandon_effect = {
        add_base_tax = -1
    }
    modifier = {
        local_missionary_strength = 0.01
        local_religious_unity_contribution = -0.5
    }
    ai_priority = {
        factor = 2
        modifier = {
            factor = 2.0
            NOT = { religion = ROOT }
        }
    }
    localization = holy_order
}

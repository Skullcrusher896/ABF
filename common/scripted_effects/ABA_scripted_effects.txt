aba_check_urbanization_effect = {
	# ROOT is the province scope
	if = {
		limit = {
			# Condition to HAVE the modifier
			owner = { has_government_attribute = thriving_cities_modifier }
			development = 25
		}
		if = {
			limit = { NOT = { has_province_modifier = aba_thriving_cities_gov_mod } }
			add_province_modifier = {
				name = "aba_thriving_cities_gov_mod"
				duration = -1
			}
		}
	}
	else = {
		# If they don't meet the requirements anymore, strip it
		if = {
			limit = { has_province_modifier = aba_thriving_cities_gov_mod }
			remove_province_modifier = aba_thriving_cities_gov_mod
		}
	}
}

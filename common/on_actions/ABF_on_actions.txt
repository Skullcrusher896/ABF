			has_country_flag = ABA_pol_catholic_golden_liberty_ruler
		}

			id = aba_pol_events.5
		}
	}

			has_country_flag = aba_pol_commonwealth_appointed_grand_chancellor
		}

		limit = {
			has_country_flag = aba_pol_commonwealth_cooperated_with_magnates
		}
		clr_country_flag = aba_pol_commonwealth_cooperated_with_magnates
	}
	if = {
		limit = {
			has_country_flag = aba_pol_commonwealth_appointed_hetman
		}
		clr_country_flag = aba_pol_commonwealth_appointed_hetman
	}

		limit = {
			has_country_flag = ABA_bul_simeon_trait
		}
		add_ruler_personality = abf_legacy_of_simeon_personality

			has_country_flag = ABA_lva_conquerer_of_slavs

on_province_culture_converted = {
	if = {
		limit = {
			owner = {
				has_country_flag = ABA_pol_orthodox_culture_convert
			}
		}
		if = {
			limit = {
				NOT = {
					has_province_flag = ABA_pol_orthodox_converted_culture_bonus
				}
				culture = aba_polish_eastern
			}
			set_province_flag = ABA_pol_orthodox_converted_culture_bonus
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
	}
	if = {
		limit = {
			owner = {
				has_country_flag = ABA_bul_culture_convert_bonus
			}
		}
		if = {
			limit = {
				NOT = {
					has_province_flag = ABA_bul_culture_convert_prov_bonus
				}
				culture = bulgarian
			}
			set_province_flag = ABA_pol_orthodox_converted_culture_prov_bonus
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
	}
}

on_province_owner_change = {
	#POL
	if = {
		limit = {
			owner = {
				has_country_flag = aba_pol_embrace_diversity_culture_debuff_flag
			}
			NOT = {
				culture_group = west_slavic
			}
			has_owner_accepted_culture = no
		}
		add_province_modifier = {
			name = aba_pol_culture_modifier
			duration = -1
		}
	}
	#HTM
	if = {
		limit = {
			owner = {
				has_country_flag = aba_htm_tolerate_tatars_malus_buff
			}
			culture_group = tartar
			has_owner_accepted_culture = no
		}
		add_province_modifier = {
			name = aba_pol_culture_modifier
			duration = -1
		}
	}
}

on_province_culture_changed = {
	# POL
	if = {
		limit = {
			owner = {
				has_country_flag = aba_pol_embrace_diversity_culture_debuff_flag
			}
		}
		if = {
			limit = {
				NOT = { culture_group = west_slavic }
				NOT = { has_owner_accepted_culture = yes }
				NOT = { has_province_modifier = aba_pol_culture_modifier }
			}
			add_province_modifier = {
				name = aba_pol_culture_modifier
				duration = -1
			}
		}
		if = {
			limit = {
				has_province_modifier = aba_pol_culture_modifier
				OR = {
					culture_group = west_slavic
					has_owner_accepted_culture = yes
				}
			}
			remove_province_modifier = aba_pol_culture_modifier
		}
	}
	#HTM
	if = {
		limit = {
			owner = {
				has_country_flag = aba_htm_tolerate_tatars_malus_buff
			}
		}
		if = {
			limit = {
				culture_group = tartar
				NOT = { has_owner_accepted_culture = yes }
				NOT = { has_province_modifier = aba_pol_culture_modifier }
			}
			add_province_modifier = {
				name = aba_pol_culture_modifier
				duration = -1
			}
		}
		if = {
			limit = {
				has_province_modifier = aba_pol_culture_modifier
				NOT = { culture_group = tartar }	
			}
			remove_province_modifier = aba_pol_culture_modifier
		}
	}
}

on_culture_promoted = {
	# POL
	if = {
        limit = {
            has_country_flag = aba_pol_embrace_diversity_culture_debuff_flag
        }
        every_owned_province = {
            limit = {
				NOT = { culture_group = west_slavic }
				has_owner_accepted_culture = yes
                has_province_modifier = aba_pol_culture_modifier
            }
            remove_province_modifier = aba_pol_culture_modifier
        }
    }
	#HTM
	if = {
        limit = {
            has_country_flag = aba_htm_tolerate_tatars_malus_buff
        }
        every_owned_province = {
            limit = {
				culture_group = tartar 
				has_owner_accepted_culture = yes
                has_province_modifier = aba_pol_culture_modifier
            }
            remove_province_modifier = aba_pol_culture_modifier
        }
    }

}

on_battle_won_country = {
	if = {
		limit = {
			has_country_flag = ABF_pol_cossack_loyalty_bonus
			num_of_cossacks = 20
		}
		add_estate_loyalty = {
			estate = estate_cossacks
			loyalty = 3
		}
	}
	if = {
		limit = {
			has_country_flag = ABA_TCR_winning_battles_movement
		}
		remove_country_modifier = rmg_winning_battle
		add_country_modifier = {
			name = rmg_winning_battle
			duration = 30
		}
	}
	if = {
		limit = {
			OR = {
				has_mission = lva_sword_of_perkunas
				has_mission = lva_thunderer
			}
			FROM = {
				religion_group = christian
			}
		}
		change_variable = { which = ABF_num_of_won_battles_against_christians value = 1 }
	}
	if = {
		limit = {
			has_country_flag = ABF_pol_cossack_battle_win_bonus
		}
		remove_country_modifier = ABF_pol_cossack_battle
		add_country_modifier = {
			name = ABF_pol_cossack_battle
			duration = 60
		}
	}
	if = {
		limit = {
			has_country_flag = abf_pol_winged_hussars_battle_bonus
		}
		remove_country_modifier = ABF_pol_cossack_battle
		add_country_modifier = {
			name = ABF_pol_hussars_reinforce
			duration = 30
		}
	}
	if = {
		limit = {
			has_country_flag = aba_pol_winning_battles_warx_reduce
		}
		add_war_exhaustion = -0.05
	}
}

					tag = SCA
					tag = NSE
				}
			}
		}
		remove_country_modifier = ab_pol_baltic_peace
	}
	if = {
		limit = {
			FROM = {
				has_country_modifier = ab_pol_baltic_peace
			}
			OR = {
				tag = SWE
				tag = DAN
				tag = NOR
				tag = SCA
				tag = NSE
			}
		}
		FROM = {
			remove_country_modifier = ab_pol_baltic_peace
		}
	}
	if = {
		limit = {
			has_country_modifier = ab_pol_shared_sea
			FROM = {
				OR = {
					tag = SWE
					tag = DAN
					tag = NOR
					tag = SCA
					tag = NSE
				}
			}
		}
		remove_country_modifier = ab_pol_shared_sea

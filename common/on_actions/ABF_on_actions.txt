
on_startup = {
	if = {
		limit = {
			ai = no
		}
		country_event = {
			id = ABA_start.0
		}
		#HTM
		if = {
			limit = {
				OR = {
					has_government_mechanic = cossacks_mechanic
					has_government_mechanic = cossack_militarization_abilities_mechanic
				}
			}
			country_event = {
				id = ABA_start.1
			}
		}
	}
}

on_new_monarch = {

		limit = {
			has_country_flag = abf_heirs_of_rus_personality
		}
		add_ruler_personality = abf_heir_of_rus_personality

		limit = {
			has_country_flag = ABA_pol_catholic_golden_liberty_ruler

			id = aba_pol_events.5
		}
	}

		}

		limit = {
			has_country_flag = aba_pol_commonwealth_cooperated_with_magnates
		}
		clr_country_flag = aba_pol_commonwealth_cooperated_with_magnates
	}
	if = {
		limit = {
			has_country_flag = aba_pol_commonwealth_appointed_hetman
		}
		clr_country_flag = aba_pol_commonwealth_appointed_hetman
	}
	### BUL
	if = {
		limit = {
			has_country_flag = ABA_bul_simeon_trait
		}
		add_ruler_personality = abf_legacy_of_simeon_personality
	}
	#### LVA
	if = {
		limit = {
			has_country_flag = ABA_lva_conquerer_of_slavs
		}
		add_ruler_personality = abf_lva_vanquisher_of_slavs
	}
}

on_holy_order_established = {
	if = {
		limit = {
			owner = {
				has_country_flag = aba_htm_regiment_district_buff_flag
			}
			OR = {
				holy_order = aba_htm_chyhyryn_order
				holy_order = aba_htm_poltava_order
				holy_order = aba_htm_nizhyn_order
				holy_order = aba_htm_lubny_order
				holy_order = aba_htm_starodub_order
				holy_order = aba_htm_chernihiv_order
			}
			NOT = {
				has_province_modifier = aba_htm_efficient_colonels
			}
		}
		add_province_modifier = {
            name = aba_htm_efficient_colonels
            duration = -1
        }
	}
}

on_province_culture_converted = {
	#POL/LUD
	if = {
		limit = {
			owner = {
				has_country_flag = ABA_pol_orthodox_culture_convert
			}
		}
		if = {
			limit = {
				NOT = {
					has_province_flag = ABA_pol_orthodox_converted_culture_bonus
				}
				culture = aba_polish_eastern
			}
			set_province_flag = ABA_pol_orthodox_converted_culture_bonus
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
	}
	#BUL
	if = {
		limit = {
			owner = {
				has_country_flag = ABA_bul_culture_convert_bonus
			}
		}
		if = {
			limit = {
				NOT = {
					has_province_flag = ABA_bul_culture_convert_prov_bonus
				}
				culture = bulgarian
			}
			set_province_flag = ABA_pol_orthodox_converted_culture_prov_bonus
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
	}
}

on_province_owner_change = {
	#POL
	if = {
		limit = {
			owner = {
				has_country_flag = aba_pol_embrace_diversity_culture_debuff_flag
			}
			NOT = {
				culture_group = west_slavic
			}
			has_owner_accepted_culture = no
		}
		add_province_modifier = {
			name = aba_pol_culture_modifier
			duration = -1
		}
	}
	#HTM
	if = {
		limit = {
			owner = {
				has_country_flag = aba_htm_tolerate_tatars_malus_buff
			}
			culture_group = tartar
			has_owner_accepted_culture = no
		}
		add_province_modifier = {
			name = aba_pol_culture_modifier
			duration = -1
		}
	}
	if = {
		limit = {
			owner = {
				has_government_attribute = thriving_cities_modifier	
			}
			development = 25
		}
		add_province_modifier = { 
			name = aba_thriving_cities_gov_mod
			duration = -1
			hidden = yes
		}
	}
	if = {
		limit = {
			owner = {
				NOT = {
					has_government_attribute = thriving_cities_modifier	
				}
			}
			has_province_modifier = aba_thriving_cities_gov_mod 
		}
		remove_province_modifier = aba_thriving_cities_gov_mod 
	}
	if = {
		limit = {
			owner = {
				has_country_flag = aba_htm_regiment_district_buff_flag
			}
		}
		aba_check_buff_regiment_effect = yes
	}
	if = {
		limit = {
			owner = {
				NOT = {
					has_country_flag = aba_htm_regiment_district_buff_flag
				}
			}
			has_province_modifier = aba_htm_efficient_colonels
		}
		remove_province_modifier = aba_htm_efficient_colonels
	}
	#ZUN
	if = {
		limit = {
			owner = {
				has_country_flag = aba_znn_burmese_integration
			}
			culture_group = burman
			has_owner_accepted_culture = no
		}
		add_province_modifier = {
			name = aba_pol_culture_modifier
			duration = -1
		}
	}
}

on_battle_won_country = {
	#POL
	if = {
		limit = {
			has_country_flag = ABF_pol_cossack_loyalty_bonus
			num_of_cossacks = 20
		}
		add_estate_loyalty = {
			estate = estate_cossacks
			loyalty = 3
		}
	}
	if = {
		limit = {
			has_country_flag = ABF_pol_cossack_battle_win_bonus
		}
		remove_country_modifier = ABF_pol_cossack_battle
		add_country_modifier = {
			name = ABF_pol_cossack_battle
			duration = 60
		}
	}
	if = {
		limit = {
			has_country_flag = abf_pol_winged_hussars_battle_bonus
		}
		remove_country_modifier = ABF_pol_cossack_battle
		add_country_modifier = {
			name = ABF_pol_hussars_reinforce
			duration = 30
		}
	}
	if = {
		limit = {
			has_country_flag = aba_pol_winning_battles_warx_reduce
		}
		add_war_exhaustion = -0.05
	}
	#TCR
	if = {
		limit = {
			has_country_flag = ABA_TCR_winning_battles_movement
		}
		remove_country_modifier = rmg_winning_battle
		add_country_modifier = {
			name = rmg_winning_battle
			duration = 30
		}
	}
	#LVA
	if = {
		limit = {
			OR = {
				has_mission = lva_sword_of_perkunas
				has_mission = lva_thunderer
			}
			FROM = {
				religion_group = christian
			}
		}
		change_variable = { which = ABF_num_of_won_battles_against_christians value = 1 }
	}
	#HTM
	if = {
		limit = {
			has_country_flag = aba_htm_free_slaves_battles
			FROM = {
				government = tribal
			}
		}
		add_yearly_manpower = 0.1
	}
	if = {
		limit = {
			has_country_flag = aba_htm_winning_battles_monarchies_rtradition
			FROM = {
				government = monarchy
			}
		}
		add_republican_tradition = 0.05
	}
}

on_monthly_pulse = {
    #HTM/COSSACK MECHANIC --- FIRST MECHANIC: FLEEING SERFS TARGETING ---
    if = {
        limit = {
            OR = {
                has_government_mechanic = cossacks_mechanic
                has_government_mechanic = cossack_militarization_abilities_mechanic 
            }
        }
        
        clear_global_event_target = htm_cossack_current_serf_target
        
        random_rival_country = {
            limit = { is_neighbor_of = ROOT }
            save_global_event_target_as = htm_cossack_current_serf_target
        }
        
        # Fallback: If no rival was found, pick a random neighbor country
        if = {
            limit = {
                NOT = { has_saved_global_event_target = htm_cossack_current_serf_target }
            }
            random_neighbor_country = {
                save_global_event_target_as = htm_cossack_current_serf_target
            }
        }
    }

    #HTM/COSSACK MECHANIC --- SECOND MECHANIC: RAIDING TARGETING ---
    if = {
        limit = {
            has_country_flag = aba_htm_raiding_parties_interaction_buff 
            OR = {
                has_government_mechanic = cossacks_mechanic
                has_government_mechanic = cossack_militarization_abilities_mechanic 
            }
        }
        
        clear_global_event_target = htm_cossack_current_raid_target
		clear_global_event_target = htm_cossack_province_raid_target
        
        random_war_enemy_country = {
            limit = {
                is_neighbor_of = ROOT
                NOT = { tag = event_target:htm_cossack_current_serf_target }
            }
            save_global_event_target_as = htm_cossack_current_raid_target
        }
        
        # Fallback: If not at war with a neighbor, pick an unallied neighbor
        if = {
            limit = {
                NOT = { has_saved_global_event_target = htm_cossack_current_raid_target }
            }
            random_neighbor_country = {
                limit = {
                    NOT = { tag = event_target:htm_cossack_current_serf_target }
                    NOT = {
                        has_opinion = {
                            who = ROOT
                            value = 0
                        }
                    }
                }
                save_global_event_target_as = htm_cossack_current_raid_target
            }
        }

		# PROVINCE LOCK: Now that a valid raid target country is guaranteed, 
        # look inside them, pick ONE random border province, and lock it down for the month.
        if = {
            limit = {
                has_saved_global_event_target = htm_cossack_current_raid_target
            }
            event_target:htm_cossack_current_raid_target = {
                random_owned_province = {
                    limit = {
                        OR = {
                            is_neighbor_of = HTM
                            is_neighbor_of = ZAZ
                        }
                    }
                    save_global_event_target_as = htm_cossack_province_raid_target
                }
            }
        }
    }
}

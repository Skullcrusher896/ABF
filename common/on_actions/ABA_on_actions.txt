
#Hindusthani Assimilation Bonus - Shahi
on_province_owner_change = {
SHH = {
    if = {
      limit = {
        NOT = {
        any_province = { 
            culture_group = hindusthani
            NOT = { 
                owned_by = ROOT 
                }
            }
        }
      }
      add_country_modifier = {
        name = ABA_SHH_hindusthani_assimilated
        duration = -1
      }
  }
  else_if = {
  limit = {
    any_province = {
      culture_group = hindusthani
      NOT = { owned_by = ROOT }
    }
    has_country_modifier = ABA_SHH_hindusthani_assimilated
  }
  remove_country_modifier = ABA_SHH_hindusthani_assimilated
   }
  }
}

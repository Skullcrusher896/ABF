namespace = ABA_start

# Startup Setup
country_event = {
	id = ABA_start.0
	title = ABA_start.0.t
	desc = ABA_start.0.d
	picture = CULTURE_eventPicture  
	
	hidden = yes
	fire_only_once = yes
	is_triggered_only = yes

    option = {
        name = ABA_start.0.a
		LVA = {
			set_variable = {
				which = ABF_num_of_won_battles_against_christians
				value = 0
			}
		}
		PLN = {
			set_country_flag = aba_pln_crownpoland_popular
		}
    }
}

# Cossack Mechanic events
country_event = {
	id = ABA_start.1
	title = ABA_start.1.t
	desc = ABA_start.1.d
	picture = CULTURE_eventPicture  
	
	hidden = yes
	fire_only_once = yes
	is_triggered_only = yes

	immediate = {
		# --- INITIALIZE FLEEING SERFS TARGET ---
		clear_global_event_target = htm_cossack_current_serf_target
		
		random_rival_country = {
			limit = { is_neighbor_of = ROOT }
			save_global_event_target_as = htm_cossack_current_serf_target
		}
		
		if = {
			limit = {
				NOT = { has_saved_global_event_target = htm_cossack_current_serf_target }
			}
			random_neighbor_country = {
				save_global_event_target_as = htm_cossack_current_serf_target
			}
		}

		# --- INITIALIZE RAIDING TARGETS ---
		if = {
			limit = {
				has_country_flag = aba_htm_raiding_parties_interaction_buff
			}
			clear_global_event_target = htm_cossack_current_raid_target
			clear_global_event_target = htm_cossack_province_raid_target
			
			random_war_enemy_country = {
				limit = {
					is_neighbor_of = ROOT
					NOT = { tag = event_target:htm_cossack_current_serf_target }
				}
				save_global_event_target_as = htm_cossack_current_raid_target
			}
			
			if = {
				limit = {
					NOT = { has_saved_global_event_target = htm_cossack_current_raid_target }
				}
				random_neighbor_country = {
					limit = {
						NOT = { tag = event_target:htm_cossack_current_serf_target }
						NOT = {
							has_opinion = {
								who = ROOT
								value = 0
							}
						}
					}
					save_global_event_target_as = htm_cossack_current_raid_target
				}
			}
			
			# Lock down the initial starting province
			if = {
				limit = {
					has_saved_global_event_target = htm_cossack_current_raid_target
				}
				event_target:htm_cossack_current_raid_target = {
					random_owned_province = {
						limit = {
							OR = {
								is_neighbor_of = HTM
								is_neighbor_of = ZAZ
							}
						}
						save_global_event_target_as = htm_cossack_province_raid_target
					}
				}
			}
		}
	}

    option = {
        name = ABA_start.1.a
    }
}

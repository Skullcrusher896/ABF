	is_triggered_only = yes

	option = { #Integrate the Marathas
		name = aba_zun_event.7.a
		custom_tooltip = ABA_ZNN_MARATHAS_PRIVILEGE_TT
		hidden_effect = {
			unlock_estate_privilege = {
				estate_privilege = ABA_ZNN_estate_nobles_maratha_special_privilege
			}
		}
	}

	option = { #Our current class of warrior-nobility is enough
		name = aba_zun_event.7.b
		custom_tooltip = AB_SHH_improved_nobles_privilege_tt
		hidden_effect = {
			if = {
				limit = {
					has_estate_privilege = estate_nobles_command_of_the_military
				}
				remove_estate_privilege = estate_nobles_command_of_the_military
				set_country_flag = AB_SHH_improved_nobles_privilege
				set_estate_privilege = AB_estate_nobles_command_of_the_military_improved
			}
			else = {
				set_country_flag = AB_improved_command_military_privilege
			}
		}
	}
}

	title = aba_zun_event.8.t
	desc = aba_zun_event.8.d
	picture = LAND_MILITARY_eventPicture
	
	is_triggered_only = yes

	option = { #Rise warrios of the desert!
		name = aba_zun_event.8.a
		custom_tooltip = ABA_ZNN_QIZILIBASH_SPAWN_TT
		if = {
			limit = {
				num_of_qizilbash = 30
			}
			add_country_modifier = {
				name = ABA_ZNN_qizilbash_integration
				duration = -1
			}
		}
		hidden_effect = {
			capital_scope = {
				for = {
					amount = 3
					effect = "
						qizilbash_cavalry = ROOT
					"
				}
			}
			capital_scope = {
				for = {
					amount = 7
					effect = "
						qizilbash_infantry = ROOT
					"
				}
			}			
			country_event = {
				id = aba_zun_event.8
				days = 3650
			}
		}
	}	
}

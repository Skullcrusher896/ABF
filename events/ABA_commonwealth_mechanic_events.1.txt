#Appoint a Grand Chancellor of the Crown
country_event = {
	id = aba_commonwealth_mechanic_events.1
	title = aba_commonwealth_mechanic_events.1.t
	desc = aba_commonwealth_mechanic_events.1.d
	picture = WESTERNISATION_eventPicture
	
	is_triggered_only = yes

	immediate = {
		set_country_flag = aba_pol_commonwealth_appointed_grand_chancellor
	}

	option = {
		name = aba_commonwealth_mechanic_events.1.a
		trigger = {
			poland_region = {
				type = any
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		if = {
			limit = {
				has_country_flag = aba_pol_catholic_adm_interaction_buff1
			}
			add_ruler_modifier = {
				name = aba_pol_polish_chancellor_buffed
				duration = -1
			}
		}
		else = {
			add_ruler_modifier = {
				name = aba_pol_polish_chancellor_default
				duration = -1
			}
		}
	}

	option = {
		name = aba_commonwealth_mechanic_events.1.b
		trigger = {
			ruthenia_region = {
				type = any
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		if = {
			limit = {
				has_country_flag = aba_pol_catholic_adm_interaction_buff2
			}
			add_ruler_modifier = {
				name = aba_pol_ruthenian_chancellor_buffed
				duration = -1
			}
		}
		else = {
			add_ruler_modifier = {
				name = aba_pol_ruthenian_chancellor_default
				duration = -1
			}
		}
	}

	option = {
		name = aba_commonwealth_mechanic_events.1.c
		trigger = {
			baltic_region = {
				type = any
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		if = {
			limit = {
				has_country_flag = aba_pol_catholic_adm_interaction_buff3
			}
			add_ruler_modifier = {
				name = aba_pol_litvin_chancellor_buffed
				duration = -1
			}
		}
		else = {
			add_ruler_modifier = {
				name = aba_pol_litvin_chancellor_default
				duration = -1
			}
		}
	}

	option = {
		name = aba_commonwealth_mechanic_events.1.f
		add_ruler_modifier = {
			name = aba_pol_russian_chancellor_default
			duration = -1
		}
	}

	option = {
		name = aba_commonwealth_mechanic_events.1.e
		trigger = {
			has_country_flag = aba_pol_commonwealth_tartar_access
		}
		add_ruler_modifier = {
			name = aba_pol_tartar_chancellor_default
			duration = -1
		}
	}
}

#Cooperate with Magnates
country_event = {
	id = aba_commonwealth_mechanic_events.2
	title = aba_commonwealth_mechanic_events.2.t
	desc = aba_commonwealth_mechanic_events.2.d
	picture = PARLIAMENT_eventPicture
	
	is_triggered_only = yes

	immediate = {
		set_country_flag = aba_pol_commonwealth_cooperated_with_magnates
	}

	option = {
		name = aba_commonwealth_mechanic_events.2.a
		trigger = {
			poland_region = {
				type = any
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		random_list = {
			33 = {
				add_ruler_modifier = {
					name = aba_pol_potocki_magnates
					duration = -1
				}
			}
			33 = {
				add_ruler_modifier = {
					name = aba_pol_kniecpolski_magnates
					duration = -1
				}
			}
			33 = {
				if = {
					limit = {
						NOT = {
							has_country_flag = aba_pol_commonwealth_zamosc_buff
						}
					}
					add_ruler_modifier = {
						name = aba_pol_zamoyski_magnates
						duration = -1
					}
				}
				else  = {
					add_ruler_modifier = {
						name = aba_pol_zamoyski_magnates_buffed
						duration = -1
					}
				}
			}
		}
	}

					name = aba_pol_wisniowiecki_magnates
					duration = -1
				}
			}
		}
	}

	option = {
		name = aba_commonwealth_mechanic_events.2.c
		trigger = {

				type = any
				country_or_non_sovereign_subject_holds = ROOT
			}
			has_country_flag = aba_pol_catholic_dip_interaction_buff1
		}
		random_list = {
